% === 宏包加载 ===
\usepackage[UTF8]{ctex}
\usepackage{geometry}
\usepackage{fancyhdr}
\usepackage{graphicx}
\usepackage{tikz}
\usepackage{lastpage}
\usepackage{fontspec}

% === 字体设置 ===
\setmainfont{Times New Roman}
\setCJKmainfont[BoldFont=SimHei]{SimSun}
\setCJKfamilyfont{kai}{KaiTi}
\newcommand{\kaiti}{\CJKfamily{kai}}

% === 页面设置 ===
\geometry{
    left=1.27cm,
    right=1.27cm,
    top=0.43cm,
    bottom=0.44cm,
    headheight=1.2cm,
    headsep=0.3cm,
    footskip=0.8cm,
    includehead,
    includefoot
}

\setlength{\parindent}{0pt}

% === 答题卡信息 ===
\newcommand{\answersheettitletext}{第XX届CPHOS物理竞赛联考}

\newcommand{\answersheetsubtitle}[1]{\renewcommand{\answersheetsubtitletext}{#1}}

% === 页眉页脚设置 ===
\pagestyle{fancy}

\renewcommand{\footrulewidth}{0pt}

% === 答题区域环境 ===
% \begin{answerarea}{题号}...\end{answerarea}
\newenvironment{answerarea}[1]{%
    \begin{tikzpicture}[remember picture, overlay]
        \draw[line width=0.5pt] 
            ([xshift=1.27cm, yshift=-1.93cm]current page.north west) 
            rectangle 
            ([xshift=-1.27cm, yshift=1.24cm]current page.south east);
    \end{tikzpicture}%
    \vspace{0.2cm}%
    \hspace{0.3cm}{\bfseries\large #1}%
    \par\vspace{0.3cm}%
}{%
    \par%
}

% === 整页答题区域（带边框）===
% 边框坐标：左右与页眉对齐(left/right margin)，上边在页眉下方(top+headheight+headsep)，下边在页脚上方(bottom+footskip)
\newcommand{\answerpagebox}[1]{%
    \newpage
    \begin{tikzpicture}[remember picture, overlay]
        \draw[line width=0.5pt] 
            ([xshift=1.27cm, yshift=-1.93cm]current page.north west) 
            rectangle 
            ([xshift=-1.27cm, yshift=1.24cm]current page.south east);
    \end{tikzpicture}%
    \vspace{0.1cm}%
    \hspace{0.3cm}{\bfseries\large #1}%
    \par%
}
